\sectionline{Selected FHE / Applied Cryptography Projects}

\begin{itemize}[leftmargin=0em, itemsep=0.45em]
    \renewcommand\labelitemi{}

    \item
    \textbf{BGV-Based Private Information Retrieval for Hyperledger Fabric}
    \begin{itemize}[leftmargin=1.2em, itemsep=0.05em]
        \renewcommand\labelitemi{$\bullet$}
        \item Implemented a single-server HE-PIR prototype in Go using Lattigo, with packed CTI records, client-side encrypted selection vectors, and chaincode-side ciphertext-plaintext evaluation.
        \item Designed fixed-window slot packing, record-capacity checks, channel-specific parameterization, and client-side decoding for privacy-preserving reads from Fabric world state.
        \item Benchmarked end-to-end latency, peer-side evaluation time, network traffic, storage overhead, and parameter/capacity constraints; reported 113 ms end-to-end latency and $\sim$2 MB peer traffic per query in the prototype.
        \item \textbf{Stack:} Go, Lattigo, Hyperledger Fabric, Docker, BGV, PIR, RLWE-based HE.
    \end{itemize}

    \item
    \textbf{CKKS-Based Encrypted Aggregation for Federated Reinforcement Learning}
    \begin{itemize}[leftmargin=1.2em, itemsep=0.05em]
        \renewcommand\labelitemi{$\bullet$}
        \item Implemented CKKS-encrypted model-update aggregation using TenSEAL for a federated reinforcement learning network-defense prototype.
        \item Used chunking to encode model-update vectors into encrypted CKKS payloads; clients train locally in plaintext while aggregation is performed over encrypted updates.
        \item Compared plaintext, differential privacy, trust-weighted, HE, and combined HE+DAPI execution paths; focused on confidentiality of model updates rather than full encrypted training/inference.
        \item \textbf{Stack:} Python, TenSEAL, PyTorch, NumPy, CKKS, federated learning, reinforcement learning.
    \end{itemize}

\end{itemize}
